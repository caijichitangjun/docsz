    # 代码复制

    # vuepress-plugin-nuggets-style-copy

    安装：yarn add vuepress-plugin-nuggets-style-copy -D 使用:

    plugins: [
  "vuepress-plugin-nuggets-style-copy": {
    copyText: "复制代码",
    tip: {
      content: "复制成功"
    }
  }
]

    # vuepress-plugin-code-copy

    安装：yarn add -D @snippetors/vuepress-plugin-code-copy -D 使用:

    plugins: [
 "@snippetors/vuepress-plugin-code-copy": {
    // selector: 语言
    align: top/bottom  // 位置
    // color 颜色
    ... 
  }
]

    # yarn add vuepress-plugin-copyright -D

    'vuepress-plugin-code-copy': true,           // 代码复制功能
